## Phase 1 — SECRET_PATTERNS single source of truth (TDD, ONE task)

**Why this first:** the advisor flagged this as the #1 silent-correctness trap. The bash ERE uses POSIX classes (`[[:space:]]`, `[^[:space:]/:@]`) that JavaScript `RegExp` silently miscompiles to wrong character classes — `new RegExp('[[:space:]]', 'i')` compiles to a class matching `[`, `:`, `s`, `p`, etc., plus a stray `]`. The env-var and connection-string scrub categories WILL NOT match against a naive port — but `AKIA…`, `sk-ant-…`, `gh[pousr]_`, `-----BEGIN` will, so the test suite would partially pass against a broken impl. The fix: keep the bash ERE as a literal string constant pinned via fixture, build the JS RegExp via deliberate POSIX-class translation, and assert both halves.

- [ ] **Step 1.4: Implement scrub.ts (GREEN phase).**

The POSIX translation is the load-bearing piece. Translate:

- `[[:space:]]` → `\s`
- `[^[:space:]]` → `\S`
- `[^[:space:]/:@]` → `[^\s/:@]`
- `[^[:space:]/@]` → `[^\s/@]`
- `[A-Za-z0-9_]` → `\w` (equivalent for ASCII; safer to leave as-is to preserve bash ERE literal shape — translation table is below)

```typescript
// packages/cli/src/handoff/scrub.ts
//
// Cycle 12 spec §6 — secret-scrub backstop.
//
// SECRET_PATTERNS is defined in two forms:
//   - SECRET_PATTERNS_BASH_ERE — the ORIGINAL bash ERE string, kept byte-
//     identical with momentiq-ai/dark-factory-platform's lib.sh@a6f711b. This
//     is the cross-repo single source of truth (spec §6). The fixture at
//     tests/handoff/fixtures/secret-patterns.bash-ere is the cross-repo
//     bridge; scrub.test.ts asserts equality.
//   - SECRET_PATTERNS_JS — the JavaScript RegExp built from a DELIBERATE
//     POSIX-class translation (not `new RegExp(ere, 'i')`, which silently
//     miscompiles [[:space:]] etc.). This is the runtime matcher.
//
// Refusal contract (v2-tightened per spec §6): refusal NEVER echoes matched
// content — line numbers + bodyfile path only. The leaked-then-edited secret
// is still readable in the issue body's edit history.

export const SECRET_PATTERNS_BASH_ERE =
  "[A-Za-z0-9_]*(api[_-]?key|secret|token|passwd|password|access[_-]?key|private[_-]?key)[A-Za-z0-9_]*[[:space:]]*[:=][[:space:]]*[^[:space:]]" +
  "|gh[pousr]_[A-Za-z0-9]{20,}" +
  "|xox[baprs]-[A-Za-z0-9-]+" +
  "|AKIA[0-9A-Z]{16}" +
  "|sk-(ant-|proj-)?[A-Za-z0-9_-]{16,}" +
  "|AIza[A-Za-z0-9_-]{16,}" +
  "|[a-z][a-z0-9+.-]*://[^[:space:]/:@]+:[^[:space:]/@]+@" +
  "|\\.(aws/credentials|kube/config|ssh/id_[a-z0-9]+|codex/auth\\.json|config/gcloud|docker/config\\.json|netrc|npmrc|pgpass|dockercfg)" +
  "|-----BEGIN";

// VERIFY: the string above is rebuilt from lib.sh's single line. The unit test
// at scrub.test.ts asserts equality with the .bash-ere fixture. If you edit
// either, re-run the test — the fixture is the SoT.

/**
 * POSIX → JS class translation table:
 *   [[:space:]]      → \s
 *   [^[:space:]]     → \S
 *   [^[:space:]/:@]  → [^\s/:@]
 *   [^[:space:]/@]   → [^\s/@]
 *
 * Everything else in the bash ERE happens to be JS-RegExp-compatible (ASCII
 * char classes, alternations, fixed strings). The translation below is hand-
 * written rather than mechanical — it's small enough to audit by eye, and
 * the test suite covers each category behaviorally.
 */
const SECRET_PATTERNS_JS_SOURCE =
  "[A-Za-z0-9_]*(api[_-]?key|secret|token|passwd|password|access[_-]?key|private[_-]?key)[A-Za-z0-9_]*\\s*[:=]\\s*\\S" +
  "|gh[pousr]_[A-Za-z0-9]{20,}" +
  "|xox[baprs]-[A-Za-z0-9-]+" +
  "|AKIA[0-9A-Z]{16}" +
  "|sk-(ant-|proj-)?[A-Za-z0-9_-]{16,}" +
  "|AIza[A-Za-z0-9_-]{16,}" +
  "|[a-z][a-z0-9+.-]*://[^\\s/:@]+:[^\\s/@]+@" +
  "|\\.(aws/credentials|kube/config|ssh/id_[a-z0-9]+|codex/auth\\.json|config/gcloud|docker/config\\.json|netrc|npmrc|pgpass|dockercfg)" +
  "|-----BEGIN";

export const SECRET_PATTERNS_JS = new RegExp(SECRET_PATTERNS_JS_SOURCE, "i");

export interface ScrubResult {
  readonly ok: boolean;
  readonly refusal?: string;
}

/** Scrub a multi-line body. Returns line NUMBERS + bodyfile path only on match. */
export function scrubBody(body: string, bodyfile: string): ScrubResult {
  // Detection separated from line formatting so a formatting hiccup never
  // causes a MISSED secret (bash idiom — grep -q before grep -n).
  const re = new RegExp(SECRET_PATTERNS_JS.source, SECRET_PATTERNS_JS.flags);
  if (!re.test(body)) return { ok: true };
  const lines: number[] = [];
  body.split("\n").forEach((line, idx) => {
    // Per-line regex with i flag (g would advance state across exec calls).
    const lineRe = new RegExp(SECRET_PATTERNS_JS.source, SECRET_PATTERNS_JS.flags);
    if (lineRe.test(line)) lines.push(idx + 1);
  });
  const lineList = lines.length > 0 ? lines.join(",") : "?";
  return {
    ok: false,
    refusal:
      `aborted: secret-shaped content in ${bodyfile}:${lineList}; ` +
      "rephrase as setup steps (no value echo — see SKILL.md § Security rule).",
  };
}

/** Scrub a single string (e.g., a fetched PR title before splicing into body). */
export function scrubString(s: string, label: string): ScrubResult {
  const re = new RegExp(SECRET_PATTERNS_JS.source, SECRET_PATTERNS_JS.flags);
  if (!re.test(s)) return { ok: true };
  return {
    ok: false,
    refusal:
      `aborted: secret-shaped content in ${label} — ` +
      "rephrase the source (e.g. `gh pr edit <N> --title …`) so the handoff " +
      "body stays scrubable. (No value echo — see SKILL.md § Security rule.)",
  };
}

// packages/cli/src/handoff/markers.ts
//
// Marker constants + parse/splice helpers. The marker tokens are LOAD-BEARING:
// the upsert finds the existing block by them and replaces it in place,
// preserving any text outside the markers. PORT FROM dark-factory-platform
// .claude/skills/handoff/scripts/lib.sh@a6f711b lines 11-215.

export const MARKER_OPEN = "<!-- agent-context:v1 -->";
export const MARKER_CLOSE = "<!-- /agent-context:v1 -->";

/** True iff body contains an open marker preceding a close marker (first-block). */
export function validateNoteMarkers(body: string): boolean {
  const lines = body.split("\n");
  const openIdx = lines.findIndex((l) => l.includes(MARKER_OPEN));
  if (openIdx < 0) return false;
  const closeIdx = lines.findIndex((l) => l.includes(MARKER_CLOSE));
  if (closeIdx < 0) return false;
  return openIdx < closeIdx;
}

/** True iff the LAST open marker is followed by a close marker. */
export function validateLatestBlock(body: string): boolean {
  const lines = body.split("\n");
  let lastOpen = -1;
  let lastClose = -1;
  lines.forEach((l, idx) => {
    if (l.includes(MARKER_OPEN)) lastOpen = idx;
    if (l.includes(MARKER_CLOSE)) lastClose = idx;
  });
  return lastOpen >= 0 && lastClose > lastOpen;
}

/**
 * Splice an agent-context block into an existing body. If the old body
 * contains markers, FIRST-open through LAST-close is replaced by the new
 * block (this both fixes operator-error multi-blocks and is idempotent for
 * the normal one-block case). If markers absent, the new block is appended
 * with a blank-line separator. Body text outside the markers is preserved.
 */
export function spliceAgentContextBlock(oldBody: string, newBlock: string): string {
  const hasOpen = oldBody.includes(MARKER_OPEN);
  const hasClose = oldBody.includes(MARKER_CLOSE);
  if (!hasOpen || !hasClose) {
    if (oldBody.length === 0) return newBlock;
    return `${oldBody}\n\n${newBlock}`;
  }
  const lines = oldBody.split("\n");
  const out: string[] = [];
  type Mode = "pre" | "inside" | "post";
  let mode: Mode = "pre";
  for (const line of lines) {
    if (mode === "pre") {
      if (line.includes(MARKER_OPEN)) {
        out.push(newBlock);
        mode = "inside";
        continue;
      }
      out.push(line);
    } else if (mode === "inside") {
      if (line.includes(MARKER_CLOSE)) {
        mode = "post";
        continue;
      }
      // skip inside-block lines (they're replaced by newBlock)
    } else {
      // mode === "post"
      if (line.includes(MARKER_OPEN)) {
        // Another stale block — drop until next close.
        mode = "inside";
        continue;
      }
      out.push(line);
    }
  }
  return out.join("\n");
}

// packages/cli/src/handoff/args.ts
//
// Argument validators. PORT FROM dark-factory-platform lib.sh@a6f711b:
//   require_issue_number  (lines 46-53)
//   require_safe_args     (lines 61-75)
//
// requireSafeArgs is defense-in-depth: in the TS CLI, real argv kills the
// shell-metacharacter injection vector that the bash .md/$ARGUMENTS surface
// was defending against. We keep the allow-list anyway because (a) MCP tool
// callers might pass any string, and (b) it preserves byte-equivalent error
// messages with the bash impl for the ~6 ported payload-rejection tests.

import { HandoffError } from "./ports.js";

/** Strict positive-integer string → number. Empty/undefined returns undefined. */
export function requireIssueNumber(input: string | undefined): number | undefined {
  if (input === undefined || input === "") return undefined;
  // Bash: case "0|0*" rejects 0 + leading-zero. JS: same.
  if (input === "0" || /^0/.test(input)) {
    throw new HandoffError(`issue must be a positive integer (got: '${input}').`);
  }
  if (!/^[0-9]+$/.test(input)) {
    throw new HandoffError(`issue must be a positive integer (got: '${input}').`);
  }
  return Number(input);
}

/**
 * Allow-list per bash require_safe_args. Excludes shell metacharacters
 * (`;$()`'`|&><!*[]`) while permitting URL-safe chars (`?=%+~`) for
 * --link URLs like https://github.com/o/r/pull/N?tab=files.
 */
const SAFE_ARG_PATTERN = /^[a-zA-Z0-9_/#:.,@?=%+~ \-]*$/;

export function requireSafeArgs(args: readonly string[]): void {
  for (const arg of args) {
    if (!SAFE_ARG_PATTERN.test(arg)) {
      throw new HandoffError(
        "argument contains disallowed characters: refusing for safety " +
          "(allowed: alphanumeric / # : . , @ - _ ? = % + ~ space).",
      );
    }
  }
}

import { HandoffError, type GhClient } from "./ports.js";
import { MARKER_OPEN, MARKER_CLOSE } from "./markers.js";

const HANDOFF_LABEL = "handoff";

export interface LinkRefCanonical {
  /** "pr" | "issue" | "" (unknown — caller treats as "any kind") */
  readonly kind: "" | "pr" | "issue";
  /** Display form: "#N" (same-repo) or "owner/repo#N" (cross-repo). */

export interface LinkRefResolved {
  readonly kind: "pr" | "issue";
  readonly display: string;
  readonly title: string;

/** Canonicalize a link ref to (kind, display) WITHOUT any gh fetch. */
export function canonicalizeLinkRef(input: string): LinkRefCanonical {
  let kind: "" | "pr" | "issue" = "";
  let ref = input;
  if (ref.startsWith("pr:")) {
    kind = "pr";
    ref = ref.slice(3);
  } else if (ref.startsWith("issue:")) {
    kind = "issue";
    ref = ref.slice(6);
  }
  // URL forms.
  const pullMatch = ref.match(/^https?:\/\/[^/]+\/([^/]+\/[^/]+)\/pull\/(\d+)/);
  if (pullMatch) {
    return { kind: "pr", display: `${pullMatch[1]}#${pullMatch[2]}` };
  }
  const issuesMatch = ref.match(/^https?:\/\/[^/]+\/([^/]+\/[^/]+)\/issues\/(\d+)/);
  if (issuesMatch) {
    return { kind: "issue", display: `${issuesMatch[1]}#${issuesMatch[2]}` };
  }
  // owner/repo#N
  if (ref.includes("#")) {
    return { kind, display: ref };
  }
  // bare number
  return { kind, display: `#${ref}` };
}

/** Resolve to (kind, display, title) WITH a gh fetch. Throws HandoffError on failure. */
export async function resolveLinkRef(
  input: string,
  gh: GhClient,
): Promise<LinkRefResolved> {
  let kind: "" | "pr" | "issue" = "";
  let ref = input;
  if (ref.startsWith("pr:")) {
    kind = "pr";
    ref = ref.slice(3);
  } else if (ref.startsWith("issue:")) {
    kind = "issue";
    ref = ref.slice(6);
  }

  // Project URLs: explicitly refused (spec §3 OQ-12.7).
  if (/^https?:\/\/[^/]+\/[^/]+\/projects\//.test(ref) || /^https?:\/\/[^/]+\/orgs\/[^/]+\/projects\//.test(ref)) {
    throw new HandoffError(
      `link ref '${input}': GitHub project-item linkage is DEFERRED to ` +
        "Phase 12.2 (spec §3 / OQ-12.7). For Phase 12.1, link PRs and issues only.",
    );
  }

  let ownerRepo = "";
  let numberStr = "";
  let display = "";

  const pullMatch = ref.match(/^https?:\/\/[^/]+\/([^/]+\/[^/]+)\/pull\/(\d+)/);
  const issuesMatch = ref.match(/^https?:\/\/[^/]+\/([^/]+\/[^/]+)\/issues\/(\d+)/);
  if (pullMatch) {
    kind = "pr";
    ownerRepo = pullMatch[1];
    numberStr = pullMatch[2];
    display = `${ownerRepo}#${numberStr}`;
  } else if (issuesMatch) {
    kind = "issue";
    ownerRepo = issuesMatch[1];
    numberStr = issuesMatch[2];
    display = `${ownerRepo}#${numberStr}`;
  } else if (ref.includes("#")) {
    [ownerRepo, numberStr] = ref.split("#", 2);
    display = `${ownerRepo}#${numberStr}`;
  } else {
    numberStr = ref;
    display = `#${numberStr}`;
  }

  if (!/^[0-9]+$/.test(numberStr)) {
    throw new HandoffError(
      `link ref '${input}' is not a number, owner/repo#N, or supported URL (pull/issues).`,
    );
  }
  if (numberStr === "0" || /^0/.test(numberStr)) {
    throw new HandoffError(
      `link ref '${input}' must reference a positive integer (got '${numberStr}').`,
    );
  }
  const num = Number(numberStr);
  const opts = ownerRepo ? { repo: ownerRepo } : undefined;

  let title = "";
  let resolvedKind: "pr" | "issue" | "" = kind;

  // PR-first lookup (unless kind already pinned to "issue").
  if (kind === "" || kind === "pr") {
    try {
      const pr = await gh.prView(num, opts);
      title = pr.title;
      resolvedKind = "pr";
    } catch {
      title = "";
    }
  }
  // Fall through to issue lookup if PR didn't resolve AND kind allows.
  if (title === "" && (kind === "" || kind === "issue")) {
    try {
      const issue = await gh.issueView(num, opts);
      // Refuse a handoff-labeled issue link target (no link-cycles).
      if (issue.labels.some((l) => l.name === HANDOFF_LABEL)) {
        throw new HandoffError(
          `refusing to link handoff issue ${display} (no link-cycles between handoff issues).`,
        );
      }
      title = issue.title;
      resolvedKind = "issue";
    } catch (err) {
      if (err instanceof HandoffError) throw err; // re-raise link-cycle refusal
      // otherwise fall through to "not found" error
    }
  }

  if (!title || !resolvedKind) {
    throw new HandoffError(
      `ref '${input}' not found as PR or Issue in ${ownerRepo || "this repo"}.`,
    );
  }
  return { kind: resolvedKind, display, title };
}

/** Extract '- (pr|issue) <ref> — <title>' lines from the LATEST marker block. */
export function extractLinkedItems(body: string): readonly string[] {
  const lines = body.split("\n");
  let lastOpen = -1;
  let lastClose = -1;
  lines.forEach((l, idx) => {
    if (l.includes(MARKER_OPEN)) lastOpen = idx;
    if (l.includes(MARKER_CLOSE)) lastClose = idx;
  });
  if (lastOpen < 0 || lastClose <= lastOpen) return [];
  const out: string[] = [];
  let inSection = false;
  for (let i = lastOpen; i <= lastClose; i++) {
    const line = lines[i];
    if (line.startsWith("**Linked work items:**")) {
      inSection = true;
      continue;
    }
    if (!inSection) continue;
    if (/^- (pr|issue) /.test(line)) {
      out.push(line);
      continue;
    }
    if (line === "_None linked._") continue;
    if (/^\s*$/.test(line)) continue;
    inSection = false;
  }
  return out;
}

/** "- pr #103 — deploy spec" */
export function formatLinkEntry(r: LinkRefResolved): string {
  return `- ${r.kind} ${r.display} — ${r.title}`;
}

// ISO-8601 normalization and parsing. PORT FROM lib.sh@a6f711b 384-410.
// GitHub's REST API returns variants: 2026-05-30T00:00:00Z, .123Z, +00:00.

export function normalizeIso(input: string): string {
  return input
    .replace(/\.[0-9]+(Z|[+-][0-9]{2}:?[0-9]{2})?$/, "$1")
    .replace(/[+-][0-9]{2}:?[0-9]{2}$/, "Z");
}

export function isoToEpoch(input: string): number | undefined {
  const norm = normalizeIso(input);
  if (!norm) return undefined;
  const t = Date.parse(norm);
  if (Number.isNaN(t)) return undefined;
  return Math.floor(t / 1000);
}

export function formatAge(epoch: number, nowEpoch: number): string {
  const diff = nowEpoch - epoch;
  if (diff < 60)     return "just now";
  if (diff < 3600)   return `${Math.floor(diff / 60)}m ago`;
  if (diff < 86400)  return `${Math.floor(diff / 3600)}h ago`;
  return `${Math.floor(diff / 86400)}d ago`;
}

import { HandoffError, type GhClient } from "./ports.js";

import { stripControlChars } from "./strip-control.js";

export interface LinkedItemDerivation {
  readonly kind: "pr" | "issue" | "?";
  readonly display: string;
  readonly title: string;
  readonly state: "OPEN" | "CLOSED" | "MERGED" | "UNREACHABLE";
  readonly annotation: string;                   // e.g. "(merged)", "(closed)", "(unreachable: ...)"

  readonly extra?: { mergeStateStatus?: string; reviewDecision?: string; checksSummary?: string; assigneesCsv?: string };

export interface RehydrateData {
  readonly issueNumber: number;
  readonly title: string;
  readonly stateLine: string;                     // "open (unassigned ...)" | "open (assigned X)" | "closed (accepted YYYY-MM-DD)"
  readonly linkedItems: readonly LinkedItemDerivation[];
  readonly linkFailures: number;
  readonly note: string | null;                   // last marker block content (incl. markers); null if none
}

export async function deriveRehydrateData(
  issueNum: number,
  gh: GhClient,
): Promise<RehydrateData> {
  const view = await gh.issueView(issueNum).catch((err) => {
    throw new HandoffError(
      `could not derive live state for #${issueNum} (gh issue view failed) — ` +
        "fix gh/network and retry; do not proceed on the note alone.",
    );
  });
  const title = stripControlChars(view.title);
  let stateLine: string;
  if (view.state === "CLOSED") {
    const day = view.closedAt ? view.closedAt.slice(0, 10) : "";
    stateLine = day ? `closed (accepted ${day})` : "closed";
  } else {
    const csv = view.assignees.map((a) => a.login).join(",");
    stateLine = csv ? `open (assigned ${csv})` : "open (unassigned — on the stack)";
  }
  // Linked items.
  const items = extractLinkedItems(view.body);
  const derivations: LinkedItemDerivation[] = [];
  let linkFailures = 0;
  for (const ln of items) {
    const der = await deriveLinkedItem(ln, gh);
    derivations.push(der);
    if (der.state === "UNREACHABLE") linkFailures++;
  }
  // Note (last block).
  const lines = view.body.split("\n");
  let lastOpen = -1; let lastClose = -1;
  lines.forEach((l, i) => { if (l.includes(MARKER_OPEN)) lastOpen = i; if (l.includes(MARKER_CLOSE)) lastClose = i; });
  let note: string | null = null;
  if (lastOpen >= 0 && lastClose > lastOpen) {
    note = lines.slice(lastOpen, lastClose + 1).join("\n");
  }
  return { issueNumber: issueNum, title, stateLine, linkedItems: derivations, linkFailures, note };
}

async function deriveLinkedItem(entryLine: string, gh: GhClient): Promise<LinkedItemDerivation> {
  // Parse "- pr|issue <display> — <title>"
  const m = entryLine.match(/^- (pr|issue) (\S+) — (.+)$/);
  if (!m) return { kind: "?", display: entryLine, title: "", state: "UNREACHABLE", annotation: "(unknown link type)" };
  const kind = m[1] as "pr" | "issue";
  const display = m[2];
  const title = stripControlChars(m[3]);
  let ownerRepo = ""; let num = display;
  if (display.startsWith("#")) { num = display.slice(1); }
  else if (display.includes("#")) { [ownerRepo, num] = display.split("#", 2); }
  const opts = ownerRepo ? { repo: ownerRepo } : undefined;
  if (kind === "pr") {
    try {
      const pr = await gh.prView(Number(num), opts);
      if (pr.state === "MERGED")     return { kind, display, title, state: "MERGED", annotation: "(merged)" };
      if (pr.state === "CLOSED")     return { kind, display, title, state: "CLOSED", annotation: "(closed)" };
      const checks = summarizeChecks(pr.statusCheckRollup);
      const checkoutHint = ownerRepo
        ? `gh pr checkout ${num} --repo ${ownerRepo}`
        : `gh pr checkout ${num}`;
      return {
        kind, display, title, state: "OPEN", annotation: "",
        checkoutHint,
        extra: { mergeStateStatus: pr.mergeStateStatus, reviewDecision: pr.reviewDecision, checksSummary: checks },
      };
    } catch {
      return { kind, display, title, state: "UNREACHABLE", annotation: "(unreachable: gh pr view failed)" };
    }
  } else {
    try {
      const issue = await gh.issueView(Number(num), opts);
      if (issue.state === "CLOSED")  return { kind, display, title, state: "CLOSED", annotation: "(closed)" };
      const csv = issue.assignees.map((a) => a.login).join(",");
      return { kind, display, title, state: "OPEN", annotation: csv ? `[open, assigned ${csv}]` : "[open]", extra: { assigneesCsv: csv } };
    } catch {
      return { kind, display, title, state: "UNREACHABLE", annotation: "(unreachable: gh issue view failed)" };
    }
  }
}

function summarizeChecks(rollup: ReadonlyArray<{ conclusion?: string; state?: string; status?: string }>): string {
  if (rollup.length === 0) return "no checks";
  const buckets = new Map<string, number>();
  for (const c of rollup) {
    const key = (c.conclusion ?? c.state ?? "UNKNOWN").toLowerCase();
    buckets.set(key, (buckets.get(key) ?? 0) + 1);
  }
  return Array.from(buckets.entries()).map(([k, v]) => `${v} ${k}`).join(", ");
}
```
